This is what must be put into the systemd file for startup procedure 


[Unit]
Description=RobotX ROS2 RPi Launch Service
After=network.target

[Service]
Type=simple
User=robotx
WorkingDirectory=/home/robotx/ros2_ws
Environment="ROS_DOMAIN_ID=0"
Environment="PYTHONUNBUFFERED=1"
ExecStart=/bin/bash -c "source /home/robotx/ros2_ws/install/setup.bash && ros2 >
Restart=on-failure
RestartSec=3

[Install]
WantedBy=multi-user.target
